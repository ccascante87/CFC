CREATE OR REPLACE FUNCTION GETCANTIDADVALORES (CODAGENCIA IN NUMBER,CODMONEDA IN VARCHAR2 ) 
   RETURN NUMBER AS MY_RESULT NUMBER;
   BEGIN   
     SELECT SUM(MONTO_VALOR)  
     INTO MY_RESULT 
     FROM MOVIMIENTOS 
     WHERE CODIGO_MONEDA = CODMONEDA
     AND CODIGO_AGENCIA= CODAGENCIA;
     RETURN MY_RESULT;
END;
